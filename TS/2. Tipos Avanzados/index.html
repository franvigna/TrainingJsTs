2. Tipos Avanzados en TypeScript

← Volver al índice

• Tipos Compuestos

Arrays, tuplas y objetos con tipos explícitos.


// Array de números
const nums: number[] = [1, 2, 3];

// Tupla: longitud fija y tipos en posición
const tupla: [string, number] = ["edad", 30];

// Objeto tipado
interface Persona { nombre: string; edad: number; }
const p: Persona = { nombre: "Ana", edad: 30 };

• Intersección (&) y Unión (|)


// Intersección
type A = { a: string };
type B = { b: number };
type C = A & B;  // { a:string; b:number; }

// Unión
type ID = string | number;
const id1: ID = "abc";
const id2: ID = 123";

• Literales y Condicionales


// Literal
type Direccion = 'arriba' | 'abajo' | 'izq' | 'der';

// Condicional (simplificado)
type IsString<T> = T extends string ? 'es string' : 'no es string';
type R1 = IsString<number>; // "no es string"
type R2 = IsString<"hola">; // "es string"

• Tipos Especiales


// any vs unknown
let a: any = 10;
let u: unknown = "algo";

// never
function fallo(msg: string): never {
  throw new Error(msg);
}

// void
function loguear(msg: string): void {
  console.log(msg);
}

• Generics


// Función genérica
function identidad<T>(arg: T): T {
  return arg;
}

const n = identidad<number>(123);
const s = identidad("texto");
