3. Funciones en TypeScript

← Volver al índice

• Tipado de parámetros y retorno


function sumar(a: number, b: number): number {
  return a + b;
}

• Funciones anónimas y flecha


// Anónima con tipo
const resta: (a: number, b: number) => number = function(a, b) {
  return a - b;
};

// Flecha con tipos y parámetro opcional
const saludar = (nombre: string, saludo: string = "Hola"): string => {
  return \`\${saludo}, \${nombre}!\`;
};

• Funciones que retornan funciones


function crearMultiplicador(factor: number) {
  return (n: number) => n * factor;
}
const dup = crearMultiplicador(2);

• Funciones de orden superior y callbacks


// Orden superior
function aplicar(
  nums: number[],
  fn: (n: number) => number
): number[] {
  return nums.map(fn);
}

// Callback
const resultado = aplicar([1,2,3], n => n * 3);
