4. Interfaces y Tipos en TypeScript

← Volver al índice

• Definir interfaces


interface Persona {
  nombre: string;
  edad: number;
}

const p: Persona = {
  nombre: "Ana",
  edad: 30
};

• Extender interfaces


interface Empleado extends Persona {
  salario: number;
}

const e: Empleado = {
  nombre: "Luis",
  edad: 28,
  salario: 50000
};

• Implementar en clases


class Cliente implements Persona {
  constructor(
    public nombre: string,
    public edad: number
  ) {}
}

const c = new Cliente("Eva", 35);

• Definir tipos con type


type ID = string | number;
const id1: ID = "abc";
const id2: ID = 123;

• Tipos literales y uniones


type Direccion = 'arriba' | 'abajo';

const d: Direccion = 'arriba';

• Extender tipos


type Contacto = Persona & { email: string };

const ct: Contacto = {
  nombre: "Mía",
  edad: 27,
  email: "mia@mail.com"
};

• Tipos condicionales y genéricos


type Elemento = T extends string ? string[] : number[];

type Rstr = Elemento<string>;  // string[]
type Rnum = Elemento<number>;  // number[]
