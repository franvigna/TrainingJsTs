5. Clases y Programación Orientada a Objetos

← Volver al índice

• Clases y constructor


class Persona {
  public nombre: string;
  constructor(nombre: string) {
    this.nombre = nombre;
  }
  saludar(): string {
    return `Hola, soy ${this.nombre}`;
  }
}

• Herencia de clases


class Empleado extends Persona {
  private salario: number;
  constructor(n: string, s: number) {
    super(n);
    this.salario = s;
  }
  obtenerSalario(): number {
    return this.salario;
  }
}

• Miembros estáticos


class Matematica {
  static PI: number = 3.1416;
  static cuadrado(x: number): number {
    return x * x;
  }
}

• Encapsulamiento y acceso


class Cuenta {
  private saldo: number;
  constructor(initial: number) {
    this.saldo = initial;
  }
  get Saldo(): number {
    return this.saldo;
  }
  set Depositar(monto: number) {
    this.saldo += monto;
  }
}

• Interfaces y clases abstractas


interface Operable {
  operar(a: number, b: number): number;
}

class Calculadora implements Operable {
  operar(a: number, b: number): number {
    return a + b;
  }
}

abstract class Animal {
  constructor(public nombre: string) {}
  abstract emitirSonido(): void;
}

class Perro extends Animal {
  emitirSonido() { console.log("¡Guau!"); }
}
