7. Manejo de Errores

← Volver al índice

1. try / catch


try {
  JSON.parse("{"badJson": });
} catch(e) {
  console.error(e);
}

2. throw


function validarEdad(edad) {
  if(edad < 18) {
    throw new Error("Debes ser mayor de edad");
  }
  return "Acceso permitido";
}

3. finally


let recursoAbierto = false;
try {
  recursoAbierto = true;
  throw new Error("Algo falló");
} catch(e) {
  console.error(e);
} finally {
  recursoAbierto = false;
}

4. Error Objects


try {
  null.metodo();
} catch(e) {
  if(e instanceof TypeError) {
    console.error("TypeError capturado", e);
  }
}
